2546/*
2547 * Show the more-prompt and handle the user response.
2548 * This takes care of scrolling back and displaying previously displayed text.
2549 * When at hit-enter prompt "typed_char" is the already typed character,
2550 * otherwise it's NUL.
2551 * Returns TRUE when jumping ahead to "confirm_msg_tail".
2552 */
2553static int do_more_prompt(int typed_char)
2554{
2555 static bool_Bool entered = false0;
2556 int used_typed_char = typed_char;
2557 int oldState = State;
2558 int c;
2559 int retval = FALSE0;
2560 int toscroll;
2561 bool_Bool to_redraw = false0;
2562 msgchunk_T *mp_last = NULL((void*)0);
2563 msgchunk_T *mp;
2564 int i;
2565
2566 // If headless mode is enabled and no input is required, this variable
2567 // will be true. However If server mode is enabled, the message "--more--"
2568 // should be displayed.
2569 bool_Bool no_need_more = headless_mode && !embedded_mode;
2570
2571 // We get called recursively when a timer callback outputs a message. In
2572 // that case don't show another prompt. Also when at the hit-Enter prompt
2573 // and nothing was typed.
2574 if (no_need_more || entered || (State == HITRETURN(0x200 + 0x01) && typed_char == 0)) {
2575 return false0;
2576 }
2577 entered = true1;
2578
2579 if (typed_char == 'G') {
2580 /* "g<": Find first line on the last page. */
2581 mp_last = msg_sb_start(last_msgchunk);
2582 for (i = 0; i < Rows - 2 && mp_last != NULL((void*)0)
2583 && mp_last->sb_prev != NULL((void*)0); ++i)
2584 mp_last = msg_sb_start(mp_last->sb_prev);
2585 }
2586
2587 State = ASKMORE0x300;
2588 setmouse();
2589 if (typed_char == NUL'\000')
2590 msg_moremsg(FALSE0);
2591 for (;; ) {
2592 /*
2593 * Get a typed character directly from the user.
2594 */
2595 if (used_typed_char != NUL'\000') {
2596 c = used_typed_char; /* was typed at hit-enter prompt */
2597 used_typed_char = NUL'\000';
2598 } else {
2599 c = get_keystroke(resize_events);
2600 }
2601
2602
2603 toscroll = 0;
2604 switch (c) {
2605 case BS'\010': /* scroll one line back */
2606 case K_BS(-(('k') + ((int)('b') << 8))):
2607 case 'k':
2608 case K_UP(-(('k') + ((int)('u') << 8))):
2609 toscroll = -1;
2610 break;
2611
2612 case CAR'\015': /* one extra line */
2613 case NL'\012':
2614 case 'j':
2615 case K_DOWN(-(('k') + ((int)('d') << 8))):
2616 toscroll = 1;
2617 break;
2618
2619 case 'u': /* Up half a page */
2620 toscroll = -(Rows / 2);
2621 break;
2622
2623 case 'd': /* Down half a page */
2624 toscroll = Rows / 2;
2625 break;
2626
2627 case 'b': /* one page back */
2628 case K_PAGEUP(-(('k') + ((int)('P') << 8))):
2629 toscroll = -(Rows - 1);
2630 break;
2631
2632 case ' ': /* one extra page */
2633 case 'f':
2634 case K_PAGEDOWN(-(('k') + ((int)('N') << 8))):
2635 case K_LEFTMOUSE(-((253) + ((int)(KE_LEFTMOUSE) << 8))):
2636 toscroll = Rows - 1;
2637 break;
2638
2639 case 'g': /* all the way back to the start */
2640 toscroll = -999999;
2641 break;
2642
2643 case 'G': /* all the way to the end */
2644 toscroll = 999999;
2645 lines_left = 999999;
2646 break;
2647
2648 case ':': /* start new command line */
2649 if (!confirm_msg_used) {
2650 /* Since got_int is set all typeahead will be flushed, but we
2651 * want to keep this ':', remember that in a special way. */
2652 typeahead_noflush(':');
2653 cmdline_row = Rows - 1; /* put ':' on this line */
2654 skip_redraw = TRUE1; /* skip redraw once */
2655 need_wait_return = FALSE0; /* don't wait in main() */
2656 }
2657 FALLTHROUGH__attribute__((fallthrough));
2658 case 'q': // quit
2659 case Ctrl_C3:
2660 case ESC'\033':
2661 if (confirm_msg_used) {
2662 /* Jump to the choices of the dialog. */
2663 retval = TRUE1;
2664 } else {
2665 got_int = TRUE1;
2666 quit_more = TRUE1;
2667 }
2668 /* When there is some more output (wrapping line) display that
2669 * without another prompt. */
2670 lines_left = Rows - 1;
2671 break;
2672
2673 case K_EVENT(-((253) + ((int)(KE_EVENT) << 8))):
2674 // only resize_events are processed here
2675 // Attempt to redraw the screen. sb_text doesn't support reflow
2676 // so this only really works for vertical resize.
2677 multiqueue_process_events(resize_events);
2678 to_redraw = true1;
2679 break;
2680
2681 default: /* no valid response */
2682 msg_moremsg(TRUE1);
2683 continue;
2684 }
2685
2686 // code assumes we only do one at a time
2687 assert((toscroll == 0) || !to_redraw)((void) sizeof (((toscroll == 0) || !to_redraw) ? 1 : 0), __extension__
({ if ((toscroll == 0) || !to_redraw) ; else __assert_fail (
"(toscroll == 0) || !to_redraw", "../src/nvim/message.c", 2687
, __extension__ __PRETTY_FUNCTION__); }));
2688
2689 if (toscroll != 0 || to_redraw) {
2690 if (toscroll < 0 || to_redraw) {
2691 // go to start of last line
2692 if (mp_last == NULL((void*)0)) {
2693 mp = msg_sb_start(last_msgchunk);
2694 } else if (mp_last->sb_prev != NULL((void*)0)) {
2695 mp = msg_sb_start(mp_last->sb_prev);
2696 } else {
2697 mp = NULL((void*)0);
2698 }
2699
2700 /* go to start of line at top of the screen */
2701 for (i = 0; i < Rows - 2 && mp != NULL((void*)0) && mp->sb_prev != NULL((void*)0);
2702 ++i)
2703 mp = msg_sb_start(mp->sb_prev);
2704
2705 if (mp != NULL((void*)0) && (mp->sb_prev != NULL((void*)0) || to_redraw)) {
2706 // Find line to be displayed at top
2707 for (i = 0; i > toscroll; i--) {
2708 if (mp == NULL((void*)0) || mp->sb_prev == NULL((void*)0)) {
2709 break;
2710 }
2711 mp = msg_sb_start(mp->sb_prev);
2712 if (mp_last == NULL((void*)0))
2713 mp_last = msg_sb_start(last_msgchunk);
2714 else
2715 mp_last = msg_sb_start(mp_last->sb_prev);
2716 }
2717
2718 if (toscroll == -1 && !to_redraw) {
2719 grid_ins_lines(&msg_grid_adj, 0, 1, Rows, 0, Columns);
2720 grid_fill(&msg_grid_adj, 0, 1, 0, Columns, ' ', ' ',
2721 HL_ATTR(HLF_MSG)highlight_attr[(int)(HLF_MSG)]);
2722 // display line at top
2723 (void)disp_sb_line(0, mp);
2724 } else {
2725 // redisplay all lines
2726 // TODO(bfredl): this case is not optimized (though only concerns
2727 // event fragmentization, not unnecessary scroll events).
2728 grid_fill(&msg_grid_adj, 0, Rows, 0, Columns, ' ', ' ',
2729 HL_ATTR(HLF_MSG)highlight_attr[(int)(HLF_MSG)]);
2730 for (i = 0; mp != NULL((void*)0) && i < Rows - 1; i++) {
2731 mp = disp_sb_line(i, mp);
2732 ++msg_scrolled;
2733 }
2734 to_redraw = false0;
2735 }
2736 toscroll = 0;
2737 }
2738 } else {
2739 /* First display any text that we scrolled back. */
2740 while (toscroll > 0 && mp_last != NULL((void*)0)) {
2741 if (msg_do_throttle() && !msg_grid.throttled) {
2742 // Tricky: we redraw at one line higher than usual. Therefore
2743 // the non-flushed area is one line larger.
2744 msg_scrolled_at_flush--;
2745 msg_grid_scroll_discount++;
2746 }
2747 // scroll up, display line at bottom
2748 msg_scroll_up(true1);
2749 inc_msg_scrolled();
2750 grid_fill(&msg_grid_adj, Rows-2, Rows-1, 0, Columns, ' ', ' ',
2751 HL_ATTR(HLF_MSG)highlight_attr[(int)(HLF_MSG)]);
2752 mp_last = disp_sb_line(Rows - 2, mp_last);
2753 toscroll--;
2754 }
2755 }
2756
2757 if (toscroll <= 0) {
2758 // displayed the requested text, more prompt again
2759 grid_fill(&msg_grid_adj, Rows - 1, Rows, 0, Columns, ' ', ' ',
2760 HL_ATTR(HLF_MSG)highlight_attr[(int)(HLF_MSG)]);
2761 msg_moremsg(false0);
2762 continue;
2763 }
2764
2765 /* display more text, return to caller */
2766 lines_left = toscroll;
2767 }
2768
2769 break;
2770 }
2771
2772 // clear the --more-- message
2773 grid_fill(&msg_grid_adj, Rows - 1, Rows, 0, Columns, ' ', ' ',
2774 HL_ATTR(HLF_MSG)highlight_attr[(int)(HLF_MSG)]);
2775 redraw_cmdline = true1;
2776 clear_cmdline = false0;
2777 mode_displayed = false0;
2778
2779 State = oldState;
2780 setmouse();
2781 if (quit_more) {
2782 msg_row = Rows - 1;
2783 msg_col = 0;
2784 } else if (cmdmsg_rl) {
2785 msg_col = Columns - 1;
2786 }
2787
2788 entered = false0;
2789 return retval;
2790}

1235 curbuf->b_nwindows++;
1236 }

1238 curwin = save_curwin;
1239 curbuf = curwin->w_buffer;

1353/// Execute autocommands for "event" and file name "fname".
1354///
1355/// @param event event that occurred
1356/// @param fname filename, NULL or empty means use actual file name
1357/// @param fname_io filename to use for <afile> on cmdline,
1358/// NULL means use `fname`.
1359/// @param force When true, ignore autocmd_busy
1360/// @param group autocmd group ID or AUGROUP_ALL
1361/// @param buf Buffer for <abuf>
1362/// @param eap Ex command arguments
1363///
1364/// @return true if some commands were executed.
1365static bool_Bool apply_autocmds_group(event_T event,
1366 char_u *fname,
1367 char_u *fname_io,
1368 bool_Bool force,
1369 int group,
1370 buf_T *buf,
1371 exarg_T *eap)
1372{
1373 char_u *sfname = NULL((void*)0); // short file name
1374 char_u *tail;
1375 bool_Bool save_changed;
1376 buf_T *old_curbuf;
1377 bool_Bool retval = false0;
1378 char_u *save_sourcing_name;
1379 linenr_T save_sourcing_lnum;
1380 char_u *save_autocmd_fname;
1381 int save_autocmd_bufnr;
1382 char_u *save_autocmd_match;
1383 int save_autocmd_busy;
1384 int save_autocmd_nested;
1385 static int nesting = 0;
1386 AutoPatCmd patcmd;
1387 AutoPat *ap;
1388 char_u *save_cmdarg;
1389 long save_cmdbang;
1390 static int filechangeshell_busy = false0;
1391 proftime_T wait_time;
1392 bool_Bool did_save_redobuff = false0;
1393 save_redo_T save_redo;
1394 const bool_Bool save_KeyTyped = KeyTyped;
1395
1396 // Quickly return if there are no autocommands for this event or
1397 // autocommands are blocked.
1398 if (event == NUM_EVENTS || first_autopat[(int)event] == NULL((void*)0)
1399 || is_autocmd_blocked()) {
1400 goto BYPASS_AU;
1401 }
1402
1403 // When autocommands are busy, new autocommands are only executed when
1404 // explicitly enabled with the "nested" flag.
1405 if (autocmd_busy && !(force || autocmd_nested)) {
1406 goto BYPASS_AU;
1407 }
1408
1409 // Quickly return when immediately aborting on error, or when an interrupt
1410 // occurred or an exception was thrown but not caught.
1411 if (aborting()) {
1412 goto BYPASS_AU;
1413 }
1414
1415 // FileChangedShell never nests, because it can create an endless loop.
1416 if (filechangeshell_busy
1417 && (event == EVENT_FILECHANGEDSHELL
1418 || event == EVENT_FILECHANGEDSHELLPOST)) {
1419 goto BYPASS_AU;
1420 }
1421
1422 // Ignore events in 'eventignore'.
1423 if (event_ignored(event)) {
1424 goto BYPASS_AU;
1425 }
1426
1427 // Allow nesting of autocommands, but restrict the depth, because it's
1428 // possible to create an endless loop.
1429 if (nesting == 10) {
1430 EMSG(_("E218: autocommand nesting too deep"))emsg((char_u *)(gettext((char *)("E218: autocommand nesting too deep"
))));
1431 goto BYPASS_AU;
1432 }
1433
1434 // Check if these autocommands are disabled. Used when doing ":all" or
1435 // ":ball".
1436 if ((autocmd_no_enter && (event == EVENT_WINENTER || event == EVENT_BUFENTER))
1437 || (autocmd_no_leave
1438 && (event == EVENT_WINLEAVE || event == EVENT_BUFLEAVE))) {
1439 goto BYPASS_AU;
1440 }
1441
1442 // Save the autocmd_* variables and info about the current buffer.
1443 save_autocmd_fname = autocmd_fname;
1444 save_autocmd_bufnr = autocmd_bufnr;
1445 save_autocmd_match = autocmd_match;
1446 save_autocmd_busy = autocmd_busy;
1447 save_autocmd_nested = autocmd_nested;
1448 save_changed = curbuf->b_changed;
1449 old_curbuf = curbuf;
1450
1451 // Set the file name to be used for <afile>.
1452 // Make a copy to avoid that changing a buffer name or directory makes it
1453 // invalid.
1454 if (fname_io == NULL((void*)0)) {
1455 if (event == EVENT_COLORSCHEME || event == EVENT_COLORSCHEMEPRE
1456 || event == EVENT_OPTIONSET) {
1457 autocmd_fname = NULL((void*)0);
1458 } else if (fname != NULL((void*)0) && !ends_excmd(*fname)) {
1459 autocmd_fname = fname;
1460 } else if (buf != NULL((void*)0)) {
1461 autocmd_fname = buf->b_ffname;
1462 } else {
1463 autocmd_fname = NULL((void*)0);
1464 }
1465 } else {
1466 autocmd_fname = fname_io;
1467 }
1468 if (autocmd_fname != NULL((void*)0)) {
1469 // Allocate MAXPATHL for when eval_vars() resolves the fullpath.
1470 autocmd_fname = vim_strnsave(autocmd_fname, MAXPATHL4096);
1471 }
1472
1473 // Set the buffer number to be used for <abuf>.
1474 if (buf == NULL((void*)0)) {
1475 autocmd_bufnr = 0;
1476 } else {
1477 autocmd_bufnr = buf->b_fnumhandle;
1478 }
1479
1480 // When the file name is NULL or empty, use the file name of buffer "buf".
1481 // Always use the full path of the file name to match with, in case
1482 // "allow_dirs" is set.
1483 if (fname == NULL((void*)0) || *fname == NUL'\000') {
1484 if (buf == NULL((void*)0)) {
1485 fname = NULL((void*)0);
1486 } else {
1487 if (event == EVENT_SYNTAX) {
1488 fname = buf->b_p_syn;
1489 } else if (event == EVENT_FILETYPE) {
1490 fname = buf->b_p_ft;
1491 } else {
1492 if (buf->b_sfname != NULL((void*)0)) {
1493 sfname = vim_strsave(buf->b_sfname);
1494 }
1495 fname = buf->b_ffname;
1496 }
1497 }
1498 if (fname == NULL((void*)0)) {
1499 fname = (char_u *)"";
1500 }
1501 fname = vim_strsave(fname); // make a copy, so we can change it
1502 } else {
1503 sfname = vim_strsave(fname);
1504 // Don't try expanding the following events.
1505 if (event == EVENT_CMDLINECHANGED || event == EVENT_CMDLINEENTER
1506 || event == EVENT_CMDLINELEAVE || event == EVENT_CMDWINENTER
1507 || event == EVENT_CMDWINLEAVE || event == EVENT_CMDUNDEFINED
1508 || event == EVENT_COLORSCHEME || event == EVENT_COLORSCHEMEPRE
1509 || event == EVENT_DIRCHANGED || event == EVENT_FILETYPE
1510 || event == EVENT_FUNCUNDEFINED || event == EVENT_OPTIONSET
1511 || event == EVENT_QUICKFIXCMDPOST || event == EVENT_QUICKFIXCMDPRE
1512 || event == EVENT_REMOTEREPLY || event == EVENT_SPELLFILEMISSING
1513 || event == EVENT_SYNTAX || event == EVENT_SIGNAL
1514 || event == EVENT_TABCLOSED || event == EVENT_WINCLOSED) {
1515 fname = vim_strsave(fname);
1516 } else {
1517 fname = (char_u *)FullName_save((char *)fname, false0);
1518 }
1519 }
1520 if (fname == NULL((void*)0)) { // out of memory
1521 xfree(sfname);
1522 retval = false0;
1523 goto BYPASS_AU;
1524 }
1525
1526#ifdef BACKSLASH_IN_FILENAME
1527 // Replace all backslashes with forward slashes. This makes the
1528 // autocommand patterns portable between Unix and Windows.
1529 if (sfname != NULL((void*)0)) {
1530 forward_slash(sfname);
1531 }
1532 forward_slash(fname);
1533#endif
1534
1535 // Set the name to be used for <amatch>.
1536 autocmd_match = fname;
1537
1538 // Don't redraw while doing autocommands.
1539 RedrawingDisabled++;
1540 save_sourcing_name = sourcing_name;
1541 sourcing_name = NULL((void*)0); // don't free this one
1542 save_sourcing_lnum = sourcing_lnum;
1543 sourcing_lnum = 0; // no line number here
1544
1545 const sctx_T save_current_sctx = current_sctx;
1546
1547 if (do_profiling == PROF_YES1) {
1548 prof_child_enter(&wait_time); // doesn't count for the caller itself
1549 }
1550
1551 // Don't use local function variables, if called from a function.
1552 funccal_entry_T funccal_entry;
1553 save_funccal(&funccal_entry);
1554
1555 // When starting to execute autocommands, save the search patterns.
1556 if (!autocmd_busy) {
1557 save_search_patterns();
1558 if (!ins_compl_active()) {
1559 saveRedobuff(&save_redo);
1560 did_save_redobuff = true1;
1561 }
1562 did_filetype = keep_filetype;
1563 }
1564
1565 // Note that we are applying autocmds. Some commands need to know.
1566 autocmd_busy = true1;
1567 filechangeshell_busy = (event == EVENT_FILECHANGEDSHELL);
1568 nesting++; // see matching decrement below
1569
1570 // Remember that FileType was triggered. Used for did_filetype().
1571 if (event == EVENT_FILETYPE) {
1572 did_filetype = true1;
1573 }
1574
1575 tail = path_tail(fname);
1576
1577 // Find first autocommand that matches
1578 patcmd.curpat = first_autopat[(int)event];
1579 patcmd.nextcmd = NULL((void*)0);
1580 patcmd.group = group;
1581 patcmd.fname = fname;
1582 patcmd.sfname = sfname;
1583 patcmd.tail = tail;
1584 patcmd.event = event;
1585 patcmd.arg_bufnr = autocmd_bufnr;
1586 patcmd.next = NULL((void*)0);
1587 auto_next_pat(&patcmd, false0);
1588
1589 // found one, start executing the autocommands
1590 if (patcmd.curpat != NULL((void*)0)) {
1591 // add to active_apc_list
1592 patcmd.next = active_apc_list;
1593 active_apc_list = &patcmd;
1594
1595 // set v:cmdarg (only when there is a matching pattern)
1596 save_cmdbang = (long)get_vim_var_nr(VV_CMDBANG);
1597 if (eap != NULL((void*)0)) {
1598 save_cmdarg = set_cmdarg(eap, NULL((void*)0));
1599 set_vim_var_nr(VV_CMDBANG, (long)eap->forceit);
1600 } else {
1601 save_cmdarg = NULL((void*)0); // avoid gcc warning
1602 }
1603 retval = true1;
1604 // mark the last pattern, to avoid an endless loop when more patterns
1605 // are added when executing autocommands
1606 for (ap = patcmd.curpat; ap->next != NULL((void*)0); ap = ap->next) {
1607 ap->last = false0;
1608 }
1609 ap->last = true1;
1610 check_lnums(true1); // make sure cursor and topline are valid
1611
1612 // Execute the autocmd. The `getnextac` callback handles iteration.
1613 do_cmdline(NULL((void*)0), getnextac, (void *)&patcmd,
1614 DOCMD_NOWAIT0x02 | DOCMD_VERBOSE0x01 | DOCMD_REPEAT0x04);
1615
1616 reset_lnums(); // restore cursor and topline, unless they were changed
1617
1618 if (eap != NULL((void*)0)) {
1619 (void)set_cmdarg(NULL((void*)0), save_cmdarg);
1620 set_vim_var_nr(VV_CMDBANG, save_cmdbang);
1621 }
1622 // delete from active_apc_list
1623 if (active_apc_list == &patcmd) { // just in case
1624 active_apc_list = patcmd.next;
1625 }
1626 }
1627
1628 RedrawingDisabled--;
1629 autocmd_busy = save_autocmd_busy;
1630 filechangeshell_busy = false0;
1631 autocmd_nested = save_autocmd_nested;
1632 xfree(sourcing_name);
1633 sourcing_name = save_sourcing_name;
1634 sourcing_lnum = save_sourcing_lnum;
1635 xfree(autocmd_fname);
1636 autocmd_fname = save_autocmd_fname;
1637 autocmd_bufnr = save_autocmd_bufnr;
1638 autocmd_match = save_autocmd_match;
1639 current_sctx = save_current_sctx;
1640 restore_funccal();
1641 if (do_profiling == PROF_YES1) {
1642 prof_child_exit(&wait_time);
1643 }
1644 KeyTyped = save_KeyTyped;
1645 xfree(fname);
1646 xfree(sfname);
1647 nesting--; // see matching increment above
1648
1649 // When stopping to execute autocommands, restore the search patterns and
1650 // the redo buffer. Free any buffers in the au_pending_free_buf list and
1651 // free any windows in the au_pending_free_win list.
1652 if (!autocmd_busy) {
1653 restore_search_patterns();
1654 if (did_save_redobuff) {
1655 restoreRedobuff(&save_redo);
1656 }
1657 did_filetype = false0;
1658 while (au_pending_free_buf != NULL((void*)0)) {
1659 buf_T *b = au_pending_free_buf->b_next;
1660 xfree(au_pending_free_buf);
1661 au_pending_free_buf = b;
1662 }
1663 while (au_pending_free_win != NULL((void*)0)) {
1664 win_T *w = au_pending_free_win->w_next;
1665 xfree(au_pending_free_win);
1666 au_pending_free_win = w;
1667 }
1668 }
1669
1670 // Some events don't set or reset the Changed flag.
1671 // Check if still in the same buffer!
1672 if (curbuf == old_curbuf
1673 && (event == EVENT_BUFREADPOST || event == EVENT_BUFWRITEPOST
1674 || event == EVENT_FILEAPPENDPOST || event == EVENT_VIMLEAVE
1675 || event == EVENT_VIMLEAVEPRE)) {
1676 if (curbuf->b_changed != save_changed) {
1677 need_maketitle = true1;
1678 }
1679 curbuf->b_changed = save_changed;
1680 }
1681
1682 au_cleanup(); // may really delete removed patterns/commands now
1683
1684BYPASS_AU:
1685 // When wiping out a buffer make sure all its buffer-local autocommands
1686 // are deleted.
1687 if (event == EVENT_BUFWIPEOUT && buf != NULL((void*)0)) {
1688 aubuflocal_remove(buf);
1689 }
1690
1691 if (retval == OK1 && event == EVENT_FILETYPE) {
1692 au_did_filetype = true1;
1693 }
1694
1695 return retval;
1696}

1931)

1943 arg = p;

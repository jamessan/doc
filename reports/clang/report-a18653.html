56// Count the size (in window cells) of the indent in line "ptr", with
57// 'tabstop' at "ts".
58// If @param list is TRUE, count only screen size for tabs.
59int get_indent_str(const char_u *ptr, int ts, int list)
60 FUNC_ATTR_NONNULL_ALL
61{
62 int count = 0;
63
64 for (; *ptr; ++ptr) {
65 // Count a tab for what it is worth.
66 if (*ptr == TAB'\011') {
67 if (!list || curwin->w_p_lcs_chars.tab1) {
68 // count a tab for what it is worth
69 count += ts - (count % ts);
70 } else {
71 // In list mode, when tab is not set, count screen char width
72 // for Tab, displays: ^I
73 count += ptr2cells(ptr);

77 count++;
78 } else {

599 curwin->w_cursor.col = pos->col;
600 col = pos->col;
601
602 that = get_cursor_line_ptr();
603
604 if (vi_lisp && (get_indent() == 0)) {
605 amount = 2;
606 } else {
607 char_u *line = that;
608
609 amount = 0;
610
611 while (*that && col) {
612 amount += lbr_chartabsize_adv(line, &that, (colnr_T)amount);
613 col--;
614 }
615
616 // Some keywords require "body" indenting rules (the
617 // non-standard-lisp ones are Scheme special forms):
618 // (let ((a 1)) instead (let ((a 1))
619 // (...)) of (...))
620 if (!vi_lisp && ((*that == '(') || (*that == '['))
621 && lisp_match(that + 1)) {
622 amount += 2;
623 } else {
624 that++;
625 amount++;
626 firsttry = amount;
627
628 while (ascii_iswhite(*that)) {
629 amount += lbr_chartabsize(line, that, (colnr_T)amount);
630 that++;
631 }
632
633 if (*that && (*that != ';')) {
634 // Not a comment line.
635 // Test *that != '(' to accommodate first let/do
636 // argument if it is more than one line.
637 if (!vi_lisp && (*that != '(') && (*that != '[')) {
638 firsttry++;
639 }
640
641 parencount = 0;
642 quotecount = 0;
643
644 if (vi_lisp || ((*that != '"') && (*that != '\'')
645 && (*that != '#') && ((*that < '0') || (*that > '9')))) {
646 while (*that
647 && (!ascii_iswhite(*that) || quotecount || parencount)
648 && (!((*that == '(' || *that == '[')
649 && !quotecount && !parencount && vi_lisp))) {
650 if (*that == '"') {
651 quotecount = !quotecount;
652 }
653 if (((*that == '(') || (*that == '[')) && !quotecount) {
654 parencount++;
655 }
656 if (((*that == ')') || (*that == ']')) && !quotecount) {
657 parencount--;
658 }
659 if ((*that == '\\') && (*(that + 1) != NUL'\000')) {
660 amount += lbr_chartabsize_adv(line, &that, (colnr_T)amount);
661 }
662
663 amount += lbr_chartabsize_adv(line, &that, (colnr_T)amount);
664 }
665 }
666
667 while (ascii_iswhite(*that)) {
668 amount += lbr_chartabsize(line, that, (colnr_T)amount);
669 that++;
670 }
671
672 if (!*that || (*that == ';')) {
673 amount = firsttry;
674 }
675 }
676 }
677 }
678 }
679 } else {
680 amount = 0; // No matching '(' or '[' found, use zero indent.
681 }
682 curwin->w_cursor = realpos;
683
684 return amount;
685}
